USE omada8;

DELIMITER //
CREATE TRIGGER updateOrderPrice
AFTER INSERT ON payment
FOR EACH ROW
BEGIN
    UPDATE orders
    SET price = price + NEW.amount
    WHERE id = NEW.orderId;
END//
DELIMITER ;
